public class Main{
public static void main(String[] args) {

        int a = 20, b = 10;
        System.out.println("Logical operators");
        System.out.println("and :" + ((a>b)&&(a<b))); 
        System.out.println("or :" + ((a>b)||(a<b)));
        System.out.println("not :" + (!(a<b)));
       
        
        System.out.println();
}
}
